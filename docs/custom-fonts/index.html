Custom Fonts

Adding custom fonts is very straight forward. Since Snowstorm supports any vite and rollup.js plugin, we can istall the amazing vite-plugin-fonts:

npm i -S vite-plugin-fonts

And enable it in our snowstorm.config.ts:

import { SnowstormConfig } from "@snowstorm/core/server";
import { Plugin as VitePluginFonts } from "vite-plugin-fonts";

export const Config: SnowstormConfig = {
  site: {
    build: {
      vitePlugins: [
        VitePluginFonts({
          google: {
            families: ["Inter"],
          },
        }),
      ],
    },
  },
};
