Dynamic Routes

While dynamic routes are not supported at runtime, you can customize you routes on build time:

snowstorm.config.ts

import { SnowstormConfig } from "@snowstorm/core";

export const Config: SnowstormConfig = {
  site: {
    routes: {
      customRoutes: {
        "/:param": {
          page: "[param]",
          exportParams: async () => [["page1"], ["page2"]],
        },
      },
    },
  },
};
